Phase 1 — Widget copy fidelity

Root cause (audit V1/V2): build_ui_for resolves the full ui_text copy list (label, help, placeholder, empty_text — all four are valid leaves per ptr_ui_text_leaf_fields()) but forwards only copy$label to the placeholder build_ui hook (see invoke_build_ui() and build_ui_for.ptr_ph_data_source() in R/paintr-build-ui.R). The builtin hooks therefore emit bare widgets. ptr_default_ui_text() already carries rich help/placeholder copy (e.g. params$alpha$num$help, params$legend.position$text$help, layers$facet_wrap$expr$__unnamed__$placeholder) — none of it reaches the screen. upload's build_ui additionally ignores the upload$file/upload$name copy and drops the accept filter.

1.1 P0 — Forward the whole resolved copy list to build_ui hooks  Med

Files: R/paintr-build-ui.R (invoke_build_ui, build_ui_for.ptr_ph_data_source) — and nothing in paintr-server.R (ptr_setup_consumer_uis already routes through invoke_build_ui, so it inherits the fix). Why "whole list": future leaf fields then need no re-plumbing; keep label = as a separate arg for back-compat with custom hooks declaring only function(node, label, ...).

# R/paintr-build-ui.R

invoke_build_ui <- function(node, ui_text, placeholders, layer_name,
                            ns_fn, extra, ...) {
  rendered_node <- node
  rendered_node$id <- ns_fn(node$id)
  copy <- ptr_resolve_ui_text(
    "control",
    keyword = node$keyword, layer_name = layer_name, param = node$param,
    ui_text = ui_text, placeholders = placeholders
  )
  entry <- ptr_registry_lookup(node$keyword)
  if (is.null(entry) || is.null(entry$build_ui)) {
    rlang::abort(paste0("No `build_ui` hook registered for placeholder `",
                        node$keyword, "`."))
  }
  do.call(
    entry$build_ui,
    c(list(rendered_node,
           label = copy$label,
           copy  = copy),          # <-- NEW: full resolved leaf list
      extra, list(...))
  )
}

# Wrap a control in a tagList with a helpText note when `help` is a

attach_help <- function(ui, help) {
  if (is.null(help) || !nzchar(trimws(help))) return(ui)

}

ptr_register_plot <- function(output, state) {
  ptr_validate_state(state)                       # Phase 5.2
  output[[state$server_ns_fn("ptr_plot")]] <- shiny::renderPlot({
    res <- state$runtime()
    shiny::req(isTRUE(res$ok), res$plot)
    res$plot

      graphics::plot.new()                          # blank the device
      return(invisible(NULL))
    }
    res$plot
  })
  invisible(state)
}

ptr_format_runtime_message <- function(stage, msg) {
  if (is.null(msg) || !nzchar(msg)) return(msg)
  prefix <- switch(stage, complete = "Input error: ", plot = "Plot error: ", "")
  if (startsWith(msg, prefix)) return(msg)           # idempotent
  paste0(prefix, msg)
}

# R/paintr-server.R
ptr_setup_layer_panel_classes <- function(state, input, output, session) {
  ns      <- state$server_ns_fn      # for reading input ids
  ui_ns   <- state$ui_ns_fn          # for the DOM element id (matches build_ui)
  spec    <- state$input_spec
  ck_rows <- spec[spec$role == "layer_checkbox", , drop = FALSE]
  for (i in seq_len(nrow(ck_rows))) {
    local({
      input_id   <- ck_rows$input_id[i]
      layer_name <- ck_rows$layer_name[i]
      dom_id     <- ui_ns(layer_panel_content_id(layer_name))
      shiny::observeEvent(input[[ns(input_id)]], {
        session$sendCustomMessage("ptr_set_class", list(
          id  = dom_id,
          cls = "ptr-layer-disabled",
          add = !isTRUE(input[[ns(input_id)]])
        ))
      }, ignoreNULL = FALSE)
    })
  }
  invisible(state)
}

5.1 P2 — Re-enforce the copy-field whitelist at placeholder definition  Low

File: R/paintr-registry.R (validate_copy_defaults) — uses ptr_ui_text_leaf_fields() from R/paintr-copy.R = c("label","help","placeholder","empty_text"). Its current comment literally says "Field whitelist (label/tooltip/placeholder/help/...) tightens at P6 wiring" — this is that tightening (and drop the bogus tooltip from the comment).

validate_copy_defaults <- function(copy_defaults) {
  if (is.null(copy_defaults)) return(invisible(TRUE))
  if (!is.list(copy_defaults)) rlang::abort("`copy_defaults` must be a named list.")
  nms <- names(copy_defaults)
  if (is.null(nms) || any(!nzchar(nms)) || anyDuplicated(nms)) {
    rlang::abort("`copy_defaults` must have unique, non-empty names.")
  }
  unknown <- setdiff(nms, ptr_ui_text_leaf_fields())
  if (length(unknown) > 0L) {
    rlang::abort(paste0(
      "`copy_defaults` has unsupported field(s): ", paste(sort(unknown), collapse = ", "),
      ". Allowed: ", paste(ptr_ui_text_leaf_fields(), collapse = ", "), "."
    ))
  }
  for (i in seq_along(copy_defaults)) {
    v <- copy_defaults[[i]]
    if (!is.character(v) || length(v) != 1L || is.na(v)) {
      rlang::abort(paste0("`copy_defaults$", nms[[i]], "` must be a single non-NA string."))
    }
  }
  invisible(TRUE)
}

ptr_validate_state <- function(state) {
  if (!is.list(state)) rlang::abort("`state` must be a `ptr_state` list (from `ptr_server_state()`).")
  needed_reactive <- c("tree", "runtime", "extras", "extras_exprs")
  needed_other    <- c("server_ns_fn", "eval_env")        # extend to match actual fields
  miss <- setdiff(c(needed_reactive, needed_other), names(state))
  if (length(miss) > 0L) {
    rlang::abort(paste0("`state` is missing required entries: ", paste(miss, collapse = ", "), "."))
  }
  for (nm in c(needed_reactive, "server_ns_fn")) {
    if (!is.function(state[[nm]])) rlang::abort(paste0("`state$", nm, "` must be a function/reactive."))
  }
  invisible(TRUE)
}

ptr_builtin_expr_resolve_expr <- function(value, node, ...) {
  if (!is.character(value) || length(value) != 1L || !nzchar(value)) {
    rlang::abort("expr placeholder requires exactly one expression as a string.")
  }
  exprs <- tryCatch(rlang::parse_exprs(value), error = function(e)
    rlang::abort(paste0("expr placeholder: could not parse input as R expression: ",
                        conditionMessage(e))))
  if (length(exprs) != 1L) {
    rlang::abort(paste0("expr placeholder: input must contain exactly one expression, but ",
                        length(exprs), " were found."))
  }
  exprs[[1]]
}
